3) BDD : schema.sql + seed.sql
🔐 Cyber — Stockage des mots de passe (hash)
Risque : mots de passe en clair réutilisables. Protection : stockage d’une empreinte avec password_hash(), vérification avec password_verify().

db/schema.sql

-- db/schema.sql
DROP DATABASE IF EXISTS campus_helpdesk;
CREATE DATABASE campus_helpdesk CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE campus_helpdesk;

CREATE TABLE utilisateurs (
  id INT AUTO_INCREMENT PRIMARY KEY,
  nom VARCHAR(100) NOT NULL,
  email VARCHAR(150) NOT NULL UNIQUE,
  mdp_hash VARCHAR(255) NOT NULL,
  role ENUM('ETUDIANT','TECH','ADMIN') NOT NULL DEFAULT 'ETUDIANT',
  actif TINYINT(1) NOT NULL DEFAULT 1,
  created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;

4) Core : Database + Auth + Router + View

app/core/Database.php

🔐 Cyber — PDO sécurisé (erreurs + prepares)
On active les exceptions et on désactive l’émulation pour favoriser des requêtes préparées réelles.
<?php
// app/core/Database.php
declare(strict_types=1);

final class Database {
  private static ?PDO $pdo = null;

  public static function pdo(): PDO {
    if (self::$pdo === null) {
      $dsn = "mysql:host=127.0.0.1;dbname=campus_helpdesk;charset=utf8mb4";
      self::$pdo = new PDO($dsn, "root", "", [
        PDO::ATTR_ERRMODE => PDO::ERRMODE_EXCEPTION,
        PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
        PDO::ATTR_EMULATE_PREPARES => false,
      ]);
    }
    return self::$pdo;
  }
}

app/core/Auth.php

🔐 Cyber — Contrôle d’accès serveur
Risque : accès direct à une URL. Protection : requireLogin() et requireRole().
<?php
// app/core/Auth.php
declare(strict_types=1);

final class Auth {
  public static function start(): void {
    if (session_status() !== PHP_SESSION_ACTIVE) {
      session_start();
    }
  }

  public static function user(): ?array {
    return $_SESSION['user'] ?? null;
  }

  public static function requireLogin(): void {
    if (!self::user()) {
      header("Location: /campus-helpdesk/public/index.php?route=login");
      exit;
    }
  }

  public static function requireRole(array $roles): void {
    $role = ($_SESSION['user']['role'] ?? null);
    if (!$role || !in_array($role, $roles, true)) {
      http_response_code(403);
      echo "<h1>403 Interdit</h1><p>Accès refusé.</p>";
      exit;
    }
  }

  public static function logout(): void {
    $_SESSION = [];
    if (ini_get("session.use_cookies")) {
      $params = session_get_cookie_params();
      setcookie(session_name(), '', time() - 42000,
        $params["path"], $params["domain"],
        $params["secure"], $params["httponly"]
      );
    }
    session_destroy();
  }
}

final class Router {
  public static function dispatch(string $route): void {
    switch ($route) {
      case 'login':
        (new AuthController())->login();
        return;
      case 'logout':
        (new AuthController())->logout();
        return;
      case 'dashboard':
        (new DashboardController())->dashboard();
        return;
      default:
        (new DashboardController())->home();
        return;
    }
  }
}
5) MVC : Model + Controllers + Views (S0)

app/models/UserModel.php

🔐 Cyber — Injection SQL
Protection : requête préparée PDO (prepare + execute).
<?php
// app/models/UserModel.php
declare(strict_types=1);

require_once __DIR__ . '/../core/Database.php';

final class UserModel {
  public function findByEmail(string $email): ?array {
    $stmt = Database::pdo()->prepare(
      "SELECT id, nom, email, mdp_hash, role, actif
       FROM utilisateurs
       WHERE email = :email
       LIMIT 1"
    );
    $stmt->execute(['email' => $email]);
    $u = $stmt->fetch();
    return $u ?: null;
  }
}

app/controllers/AuthController.php

🔐 Cyber — Validation + hash + session fixation
Après login : session_regenerate_id(true) empêche la réutilisation d’un identifiant de session imposé.
<?php
// app/controllers/AuthController.php
declare(strict_types=1);

require_once __DIR__ . '/../core/Auth.php';
require_once __DIR__ . '/../core/View.php';
require_once __DIR__ . '/../models/UserModel.php';

final class AuthController {

  public function login(): void {
    Auth::start();
    $error = null;

    if ($_SERVER['REQUEST_METHOD'] === 'POST') {
      // 🔐 Cyber — filtration/validation d'entrée
      $email = filter_input(INPUT_POST, 'email', FILTER_SANITIZE_EMAIL) ?? '';
      $password = (string)($_POST['password'] ?? '');

      if (!filter_var($email, FILTER_VALIDATE_EMAIL)) {
        $error = "Email invalide.";
      } elseif ($password === '') {
        $error = "Mot de passe requis.";
      } else {
        $user = (new UserModel())->findByEmail(trim($email));

        if (!$user || (int)$user['actif'] !== 1) {
          $error = "Identifiants incorrects.";
        } elseif (!password_verify($password, $user['mdp_hash'])) {
          // 🔐 Cyber — vérification via hash
          $error = "Identifiants incorrects.";
        } else {
          // 🔐 Cyber — anti session fixation
          session_regenerate_id(true);

          $_SESSION['user'] = [
            'id' => (int)$user['id'],
            'nom' => $user['nom'],
            'email' => $user['email'],
            'role' => $user['role'],
          ];

          header("Location: /campus-helpdesk/public/index.php?route=dashboard");
          exit;
        }
      }
    }

    View::render('auth/login', [
      'title' => 'Connexion',
      'error' => $error
    ]);
  }

  public function logout(): void {
    Auth::start();
    Auth::logout();
    header("Location: /campus-helpdesk/public/index.php?route=login");
    exit;
  }
}

app/controllers/DashboardController.php

<?php
// app/controllers/DashboardController.php
declare(strict_types=1);

require_once __DIR__ . '/../core/Auth.php';
require_once __DIR__ . '/../core/View.php';

final class DashboardController {

  public function home(): void {
    Auth::start();
    if (Auth::user()) {
      header("Location: /campus-helpdesk/public/index.php?route=dashboard");
      exit;
    }
    header("Location: /campus-helpdesk/public/index.php?route=login");
    exit;
  }

  public function dashboard(): void {
    Auth::start();
    Auth::requireLogin();

    $u = Auth::user();
    $role = $u['role'];

    if ($role === 'ADMIN') {
      View::render('dashboards/admin', ['title' => 'Dashboard Admin', 'user' => $u]);
      return;
    }
    if ($role === 'TECH') {
      View::render('dashboards/tech', ['title' => 'Dashboard Technicien', 'user' => $u]);
      return;
    }
    View::render('dashboards/user', ['title' => 'Dashboard Étudiant', 'user' => $u]);
  }
}
